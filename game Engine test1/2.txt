void render_scherm() {

	Rendered_Pixels temp_ptr_rendered_pixels;
	Rendered_Vertex temp_ptr_rendered_vertexes;
	Coordinate Phit;
	float Pdistance;
	int temp = 0;


	for (int y = 0; y < screen_size_y; y++) {
		angle_y[y] = (((float)y / screen_size_y) - 0.5) * pov_y;
	}


	for (int x = 0; x < screen_size_x; x++) {
		angle_x[x] = (((float)x / screen_size_x) - 0.5) * pov_x;
	}


	for (int nr = 0; nr < nr_faces; nr++) {

		temp_ptr_rendered_vertexes = rendered_vertexes[nr];

		if (temp_ptr_rendered_vertexes.look_away) {
			continue;
		}

		for (int y = 0; y < screen_size_y; y++) {
			for (int x = 0; x < screen_size_x; x++) {

				//int xy = x + (y * screen_size_x); 

				//temp_ptr_rendered_pixels = rendered_pixels[xy];

				Point player;
				player.x = angle_x[x];
				player.y = angle_y[y];


				if (Calc_PointInTriangle(player, temp_ptr_rendered_vertexes.A, temp_ptr_rendered_vertexes.B, temp_ptr_rendered_vertexes.C) == false) {
					continue;
				}
				


				temp_ptr_rendered_pixels.diff_ap_x = angle_x[x] - temp_ptr_rendered_vertexes.A.x;
				temp_ptr_rendered_pixels.diff_ap_y = angle_y[y] - temp_ptr_rendered_vertexes.A.y;

				temp_ptr_rendered_pixels.diff_bp_x = angle_x[x] - temp_ptr_rendered_vertexes.B.x;
				temp_ptr_rendered_pixels.diff_bp_y = angle_y[y] - temp_ptr_rendered_vertexes.B.y;

				//temp_ptr_rendered_pixels.slobe_ap = temp_ptr_rendered_pixels.diff_ap_y / temp_ptr_rendered_pixels.diff_ap_x;
				//temp_ptr_rendered_pixels.slobe_bp = temp_ptr_rendered_pixels.diff_bp_y / temp_ptr_rendered_pixels.diff_bp_x;




				if (temp_ptr_rendered_pixels.diff_ap_x == 0) {
					temp_ptr_rendered_pixels.slobe_ap_check = true;
				}
				else {
					temp_ptr_rendered_pixels.slobe_ap = temp_ptr_rendered_pixels.diff_ap_y / temp_ptr_rendered_pixels.diff_ap_x;
				}



				if (temp_ptr_rendered_pixels.diff_bp_x == 0) {
					temp_ptr_rendered_pixels.slobe_bp_check = true;
				}
				else {
					temp_ptr_rendered_pixels.slobe_bp = temp_ptr_rendered_pixels.diff_bp_y / temp_ptr_rendered_pixels.diff_bp_x;
				}







				//same dise of quadrant


				// #1
				if (temp_ptr_rendered_vertexes.slobe_ab_check & temp_ptr_rendered_pixels.slobe_ap_check) {
					if (temp_ptr_rendered_vertexes.diff_ab_y >= 0 != temp_ptr_rendered_pixels.diff_ap_y >= 0) {
						continue;
					}

				}
				else if (temp_ptr_rendered_vertexes.slobe_ab_check) {


					if (temp_ptr_rendered_vertexes.diff_ab_y >= 0) {
						if (temp_ptr_rendered_pixels.diff_ap_x >= 0) {
							continue;
						}
					}
					else {
						if (temp_ptr_rendered_pixels.diff_ap_x <= 0) {
							continue;
						}
					}



				}
				else if (temp_ptr_rendered_pixels.slobe_ap_check) {



					if (temp_ptr_rendered_vertexes.diff_ab_x >= 0) {
						if (temp_ptr_rendered_pixels.diff_ap_y <= 0) {
							continue;
						}
					}
					else {
						if (temp_ptr_rendered_pixels.diff_ap_y >= 0) {
							continue;
						}
					}



				}
				else {

					temp = (temp_ptr_rendered_pixels.diff_ap_x * temp_ptr_rendered_vertexes.diff_ab_x) >= 0;
					if (!temp & (temp_ptr_rendered_vertexes.slobe_ab <= temp_ptr_rendered_pixels.slobe_ap)) {
						continue;
					}
					if (temp & (temp_ptr_rendered_vertexes.slobe_ab >= temp_ptr_rendered_pixels.slobe_ap)) {
						continue;
					}
				}










				// #2
				if (temp_ptr_rendered_vertexes.slobe_ac_check & temp_ptr_rendered_pixels.slobe_ap_check) {
					if (temp_ptr_rendered_vertexes.diff_ac_y >= 0 != temp_ptr_rendered_pixels.diff_ap_y >= 0) {
						continue;
					}

				}
				else if (temp_ptr_rendered_vertexes.slobe_ac_check) {


					if (temp_ptr_rendered_vertexes.diff_ac_y >= 0) {
						if (temp_ptr_rendered_pixels.diff_ap_x <= 0) {

float calc_sign(Point p1, Point p2, Point p3)
{
	return (p1.x - p3.x) * (p2.y - p3.y) - (p2.x - p3.x) * (p1.y - p3.y);
}

bool Calc_PointInTriangle(Point pt, Point v1, Point v2, Point v3)
{
	float d1, d2, d3;
	bool has_neg, has_pos;

	d1 = calc_sign(pt, v1, v2);
	d2 = calc_sign(pt, v2, v3);
	d3 = calc_sign(pt, v3, v1);

	has_neg = (d1 < 0) || (d2 < 0) || (d3 < 0);
	has_pos = (d1 > 0) || (d2 > 0) || (d3 > 0);

	return !(has_neg && has_pos);
}

/// <param name="buffer"></param>
/// <param name="max_x"></param>
/// <param name="max_y"></param>
void set_screen_buffer(char* buffer, int max_x, int max_y) {
	float distance = 0;
	for (int x = 0; x < screen_size_x & x < max_x; x++) {
		for (int y = 0; y < screen_size_y & y < max_y; y++) {

			if (pixels_Buffer[x + (((screen_size_y - 1) - y) * screen_size_x)].item == 0) {
				continue;
			}

			distance = pixels_Buffer[x + (((screen_size_y - 1) - y) * screen_size_x)].y_value;
			if (distance > sizeof(scale) - 1) {
				distance = sizeof(scale) - 2;
			}
			*(buffer + x + (y * screen_size_x)) = scale[(int)distance];
		}
	}

	return;
}


void check_input() {

	float radiant;
	float new_x = player_coordinates.x;
	float new_y = player_coordinates.y;

	float move_factor = 2;

	char c = 0;

	while (_kbhit())
	{
		c = _getch();
	}

	radiant = player_angle;


	if (c > 0) {
		switch (c) {
		case 119:
			new_x += sin(radiant) * move_factor;
			new_y += cos(radiant) * move_factor;
			break;
		case 115:
			new_x -= sin(radiant) * move_factor;
			new_y -= cos(radiant) * move_factor;
			break;
		case 100:
			new_x += sin(radiant + PI / 2) * move_factor;
			new_y += cos(radiant + PI / 2) * move_factor;
			break;
		case 97:
			new_x += sin(radiant - PI / 2) * move_factor;
			new_y += cos(radiant - PI / 2) * move_factor;
			break;
		case 75:
			player_angle -= PI / 10;

			if (player_angle < -PI) {
				player_angle += twoPI;
			}
			break;
		case 77:
			player_angle += PI / 10;
			if (player_angle > PI) {
				player_angle -= twoPI;
			}
			break;
		default:
			;;
		}
	}

	player_coordinates.x = new_x;
	player_coordinates.y = new_y;

	return;
}
